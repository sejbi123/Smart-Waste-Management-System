CREATE TABLE ROLE
(
  role_id INT NOT NULL,
  role_name INT NOT NULL,
  PRIMARY KEY (role_id)
);

CREATE TABLE REPORT
(
  report_id INT NOT NULL,
  description INT NOT NULL,
  status INT NOT NULL,
  created_at INT NOT NULL,
  PRIMARY KEY (report_id)
);

CREATE TABLE MAINTENANCE_TASK
(
  maintenanceTask_id INT NOT NULL,
  status INT NOT NULL,
  description INT NOT NULL,
  PRIMARY KEY (maintenanceTask_id)
);

CREATE TABLE USER
(
  user_id INT NOT NULL,
  name INT NOT NULL,
  email INT NOT NULL,
  password INT NOT NULL,
  role_id INT NOT NULL,
  report_id INT NOT NULL,
  maintenanceTask_id INT NOT NULL,
  PRIMARY KEY (user_id),
  FOREIGN KEY (role_id) REFERENCES ROLE(role_id),
  FOREIGN KEY (report_id) REFERENCES REPORT(report_id),
  FOREIGN KEY (maintenanceTask_id) REFERENCES MAINTENANCE_TASK(maintenanceTask_id)
);

CREATE TABLE COLLECTION
(
  collection_id INT NOT NULL,
  status INT NOT NULL,
  collection_time INT NOT NULL,
  PRIMARY KEY (collection_id)
);

CREATE TABLE BIN
(
  bin_id INT NOT NULL,
  location INT NOT NULL,
  status INT NOT NULL,
  fill_level INT NOT NULL,
  report_id INT NOT NULL,
  maintenanceTask_id INT NOT NULL,
  collection_id INT NOT NULL,
  PRIMARY KEY (bin_id),
  FOREIGN KEY (report_id) REFERENCES REPORT(report_id),
  FOREIGN KEY (maintenanceTask_id) REFERENCES MAINTENANCE_TASK(maintenanceTask_id),
  FOREIGN KEY (collection_id) REFERENCES COLLECTION(collection_id)
);

CREATE TABLE ROUTE
(
  route_id INT NOT NULL,
  date INT NOT NULL,
  status INT NOT NULL,
  collection_id INT NOT NULL,
  PRIMARY KEY (route_id),
  FOREIGN KEY (collection_id) REFERENCES COLLECTION(collection_id)
);

CREATE TABLE DISPOSAL_RECORD
(
  disposalRecord_id INT NOT NULL,
  waste_type INT NOT NULL,
  quantity INT NOT NULL,
  date INT NOT NULL,
  PRIMARY KEY (disposalRecord_id)
);

CREATE TABLE TRUCK
(
  truck_id INT NOT NULL,
  capacity INT NOT NULL,
  status INT NOT NULL,
  collection_id INT NOT NULL,
  disposalRecord_id INT NOT NULL,
  PRIMARY KEY (truck_id),
  FOREIGN KEY (collection_id) REFERENCES COLLECTION(collection_id),
  FOREIGN KEY (disposalRecord_id) REFERENCES DISPOSAL_RECORD(disposalRecord_id)
);

CREATE TABLE FACILITY
(
  facility_id INT NOT NULL,
  name INT NOT NULL,
  capacity INT NOT NULL,
  disposalRecord_id INT NOT NULL,
  PRIMARY KEY (facility_id),
  FOREIGN KEY (disposalRecord_id) REFERENCES DISPOSAL_RECORD(disposalRecord_id)
);
